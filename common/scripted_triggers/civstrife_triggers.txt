frontier_planet_slave_tick_check = {
	has_any_planet_rebellion_stopper = no
	planet_stability < 25
	has_planet_flag = frontier_planet_@root
	NOT = { planet_devastation >= 30 }
	NOT = { any_targeting_situation = { is_situation_type = situation_civstrife_sep_plan } }
	if = {
		limit = { exists = sector }
		sector = {
			NOT = { has_sector_flag = sector_sep_active }
			NOT = { has_sector_flag = sec_slave_sep_active }
		}
	}
	count_owned_pop = {
		limit = {
			can_think = yes
			pop_has_happiness = yes
		}
		count >= 10
	}
	pop_percentage = {
		percentage > 0.3
		limit = {
			is_enslaved_or_purged = yes
			happiness < 60
		}
		exclude = { can_think = no }
	}
}

			}
			exclude = { can_think = no }
		}
	}
	OR = {
		planet_stability < 30
		AND = {
			planet_crime  >= 60
			OR = {
				NOT = { exists = sector.sector_capital.leader }
				AND = {
					exists = sector.sector_capital.leader

		}

wg_civwar = {
	casus_belli = cb_event

	hide_if_no_cb = yes
	set_defender_wargoal = wg_civwar_rebel

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -300

	potential = {
		always = no # event only
	}

	possible = {
		always = yes
	}

	on_accept = {
		custom_tooltip = civil_war_won_tooltip
		add_monthly_resource_mult = {
			resource = unity
			value = @tier5materialreward
			min = @tier5materialmin
		}
	}

	on_wargoal_set = {
		random_war = {
			limit = {
				using_war_goal = {
					type = wg_civwar
					owner = root
				}
			}
			every_war_participant = {
				limit = {
					NOT = { is_war_leader = yes }
				}
				prev = {
					remove_war_participant = prev
				}
			}
		}
		from = {
			every_system_within_border = {
				limit = {
					NOT = { root = { has_claim = prev } }
				}
				add_claims = { who = root }
			}
		}
	}

	ai_weight = {
		weight = 100
	}
}

wg_planet_heir_rebellion = {
	casus_belli = cb_event

	hide_if_no_cb = yes
	set_defender_wargoal = wg_planet_heir_rebellion_rebels

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -300

	potential = {
		always = no # event only
	}
	possible = {
		always = yes
	}

	on_wargoal_set = {
		random_war = {
			limit = {
				using_war_goal = {
					type = wg_planet_heir_rebellion
					owner = root
				}
			}
			every_war_participant = {
				limit = {
					NOT = { is_war_leader = yes }
				}
				prev = {
					remove_war_participant = prev
				}
			}
		}
		from = {
			every_system_within_border = {
				limit = {
					NOT = { root = { has_claim = prev } }
				}
				add_claims = { who = root }
			}
		}
		every_system_within_border = {
			limit = {
				NOT = { from = { has_claim = prev } }
			}
			add_claims = { who = from }
		}
	}

	on_accept = {
		custom_tooltip = civil_war_won_tooltip
		add_monthly_resource_mult = {
			resource = unity
			value = @tier5materialreward
			min = @tier5materialmin
		}
	}

	ai_weight = {
		weight = 100
	}
}

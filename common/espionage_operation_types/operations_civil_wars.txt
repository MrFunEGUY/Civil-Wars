operation_civil_war = {
    categories = { op_cat_subterfuge op_cat_manipulation }
    picture = GFX_evt_open_revolt
    desc = operation_civil_war_desc
    stages = 5
    spy_power_cost = 70

    resources = {
        category = operations
        cost = {
            energy = 10000
            alloys = 1000
        }
        upkeep = {
            energy = 40
        }
    }

    potential = {
        target = {
            NOT = {
                has_country_flag = civil_war_operation_target
            }
            OR = {
            	num_sectors > 1
            	any_owned_planet = {
					num_pops >= 1
					is_colony = yes
                    solar_system = { check_civwar_capital = yes }
				}
            }
        }
    }

    allow = {
        custom_tooltip = {
            is_running_espionage_operation = no
            fail_text = operation_one_at_a_time
        }
    }

    # Testing the waters
    stage = {
        # difficulty = @diff_t0
        difficulty = @diff_t6
        icon = GFX_espionage_chapter_icon_motion
        event = civesp.001
    }
    # Securing our Allies
    stage = {
        # difficulty = @diff_t0
        difficulty = @diff_t9
        icon = GFX_espionage_chapter_icon_target
        event = civesp.002
    }
    # Taking care of the logistics
    stage = {
        # difficulty = @diff_t0
        difficulty = @diff_t8
        icon = GFX_espionage_chapter_icon_motion
        event = civesp.003
    }
    # Delivering the equipment
    stage = {
        # difficulty = @diff_t0
        difficulty = @diff_t9
        icon = GFX_espionage_chapter_icon_motion
        event = civesp.004
    }
    # Putting the pieces into place
    stage = {
        # difficulty = @diff_t0
        difficulty = @diff_t7
        icon = GFX_espionage_chapter_icon_success
        event = civesp.005
    }
    on_roll_failed = {
        standard_espionage_operation_on_roll_failed = {
            RANDOM_EVENTS = operation_random_events_generic
        }
    }
    on_create = { }
}
